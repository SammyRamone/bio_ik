53  __attribute__((always_inline)) static inline double clamp2(double v,
54  double lo,
55  double hi) {
56  if (__builtin_expect(v < lo, 0)) v = lo;
57  if (__builtin_expect(v > hi, 0)) v = hi;
58  return v;
59  }

61  public:
62  RobotInfo() {}
63  RobotInfo(moveit::core::RobotModelConstPtr model) : robot_model(model) {
64  for (auto& name : model->getVariableNames()) {
65  auto& bounds = model->getVariableBounds(name);
66 
67  VariableInfo info;
68 
69  bool bounded = bounds.position_bounded_;
70 
71  auto* joint_model =
72  model->getJointOfVariable(static_cast<int>(variables.size()));
73  if (dynamic_cast<const moveit::core::RevoluteJointModel*>(joint_model)) {
74  if (bounds.max_position_ - bounds.min_position_ >= 2 * M_PI * 0.9999) {
75  bounded = false;
76  }
77  }
78 
79  info.min = bounds.min_position_;
80  info.max = bounds.max_position_;
81 
82  info.clip_min = bounded ? info.min : -DBL_MAX;
83  info.clip_max = bounded ? info.max : +DBL_MAX;
84 
85  info.span = info.max - info.min;
86 
87  if (!(info.span >= 0 && info.span < DBL_MAX)) info.span = 1;
88 
89  info.max_velocity = bounds.max_velocity_;
90  info.max_velocity_rcp =
91  info.max_velocity > 0.0 ? 1.0 / info.max_velocity : 0.0;
92 
93  variables.push_back(info);
94  }
95 
96  for (size_t ivar = 0; ivar < model->getVariableCount(); ivar++) {
97  variable_joint_types.push_back(
98  model->getJointOfVariable(static_cast<int>(ivar))->getType());
99  }
100  }
