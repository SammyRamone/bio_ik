308 class AvoidJointLimitsGoal : public Goal {
309  public:
310  AvoidJointLimitsGoal(double weight = 1.0, bool secondary = true) {
311  weight_ = weight;
312  secondary_ = secondary;
313  }
314  virtual double evaluate(const GoalContext& context) const {
315  auto& info = context.getRobotInfo();
316  double sum = 0.0;
317  for (size_t i = 0; i < context.getProblemVariableCount(); i++) {
318  size_t ivar = context.getProblemVariableIndex(i);
319  if (info.getClipMax(ivar) == DBL_MAX) continue;
320  double d = context.getProblemVariablePosition(i) -
321  (info.getMin(ivar) + info.getMax(ivar)) * 0.5;
322  d = fmax(0.0, fabs(d) * 2.0 - info.getSpan(ivar) * 0.5);
323  d *= context.getProblemVariableWeight(i);
324  sum += d * d;
325  }
326  return sum;
327  }
