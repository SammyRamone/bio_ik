43 // executes a function in parallel on pre-allocated threads
44 class ParallelExecutor {
45  volatile bool exit_;
46  std::vector<std::thread> threads_;
47  std::function<void(size_t)> function_;
48  boost::barrier barrier_;
49 
50  public:
51  template <class FUN>
52  ParallelExecutor(size_t thread_count, const FUN& function)
53  : exit_(false),
54  threads_(thread_count),
55  function_(function),
56  barrier_(static_cast<unsigned int>(thread_count)) {
57  for (size_t i = 1; i < thread_count; ++i) {
58  std::thread t([this, i]() {
59  while (true) {
60  barrier_.wait();
61  if (exit_) break;
62  function_(i);
63  barrier_.wait();
64  if (exit_) break;
65  }
66  });
67  std::swap(t, threads_[i]);
68  }
69  }
70  ~ParallelExecutor() {
71  exit_ = true;
72  barrier_.wait();
73  for (auto& t : threads_)
74  if (t.joinable()) t.join();
75  }
76  void run() {
77  barrier_.wait();
78  function_(0);
79  barrier_.wait();
80  }
81 };

96 // runs ik on multiple threads until a stop criterion is met
97 struct IKParallel {
98  IKParams params_;
99  std::vector<std::unique_ptr<IKSolver>> solvers_;
100  std::vector<std::vector<double>> solver_solutions_;
101  std::vector<std::vector<double>> solver_temps_;
102  std::vector<int> solver_success_;
103  std::vector<double> solver_fitness_;
104  size_t thread_count_;
105  // std::vector<RobotFK_Fast> fk_; // TODO: remove
106  double timeout_;
107  bool success_;
108  std::atomic<int> finished_;
109  std::atomic<uint32_t> iteration_count_;
110  std::vector<double> result_;
111  std::unique_ptr<ParallelExecutor> parallel_executor_;
112  Problem problem_;
113  bool enable_counter_;
114  double best_fitness_;
115 
116  IKParallel(const IKParams& params)
117  : params_(params), enable_counter_(params.ros_params.enable_counter) {
118  // create solvers_
119  auto create_solver = [&params]() {
120  auto result = makeSolver(params);
121  if (!result) {
122  throw std::runtime_error("Invalid Solver Name: " +
123  params.ros_params.mode);
124  }
125  return std::move(*result);
126  };
127 
128  solvers_.emplace_back(create_solver());
129  thread_count_ = solvers_.front()->concurrency();
130  while (solvers_.size() < thread_count_)
131  solvers_.emplace_back(create_solver());
132 
133  for (size_t i = 0; i < thread_count_; ++i) solvers_[i]->thread_index_ = i;
134 
135  // while(fk_.size() < thread_count_) fk_.emplace_back(params_.robot_model);
136 
137  // init buffers
138  solver_solutions_.resize(thread_count_);
139  solver_temps_.resize(thread_count_);
140  solver_success_.resize(thread_count_);
141  solver_fitness_.resize(thread_count_);
142 
143  // create parallel executor
144  parallel_executor_.reset(new ParallelExecutor(
145  thread_count_, [this](size_t i) { solverthread(i); }));
146  }
147 
148  void initialize(const Problem& problem) {
149  problem_ = problem;
150  // for(auto& f : fk_) f.initialize(problem_.tip_link_indices);
151  }
152 
153  private:
154  void solverthread(size_t i) {
155  THREADPROFILER("thread", i);
156  COUNTERPROFILER("solver threads_");
157 
158  // initialize ik solvers_
159  {
160  BLOCKPROFILER("ik solver init");
161  solvers_[i]->initialize(problem_);
162  }
163 
164  // run solver iterations until solution found or timeout_
165  for (size_t iteration = 0;
166  (rclcpp::Clock().now().seconds() < timeout_ && finished_ == 0) ||
167  (iteration == 0 && i == 0);
168  iteration++) {
169  if (finished_) break;
170 
171  // run solver for a few steps
172  solvers_[i]->step();
173  iteration_count_++;
174  for (int it2 = 1; it2 < 4; it2++)
175  if (rclcpp::Clock().now().seconds() < timeout_ && finished_ == 0)
176  solvers_[i]->step();
177 
178  if (finished_) break;
179 
180  // get solution and check stop criterion
181  auto& result = solver_temps_[i];
182  result = solvers_[i]->getSolution();
183  auto& fk = solvers_[i]->model_;
184  fk.applyConfiguration(result);
185  bool success = solvers_[i]->checkSolution(result, fk.getTipFrames());
186  if (success) finished_ = 1;
187  solver_success_[i] = success;
188  solver_solutions_[i] = result;
189  solver_fitness_[i] =
190  solvers_[i]->computeFitness(result, fk.getTipFrames());
191 
192  if (success) break;
193  }
194 
195  finished_ = 1;
196 
197  for (auto& solver : solvers_) solver->canceled_ = true;
198  }
199 
200  public:
201  void solve() {
202  BLOCKPROFILER("solve mt");
203 
204  // prepare
205  iteration_count_ = 0;
206  result_ = problem_.initial_guess;
207  timeout_ = problem_.timeout;
208  success_ = false;
209  finished_ = 0;
210  for (auto& solver : solver_solutions_) solver = problem_.initial_guess;
211  for (auto& solver : solver_temps_) solver = problem_.initial_guess;
212  for (auto& solver : solver_success_) solver = 0;
213  for (auto& f : solver_fitness_) f = DBL_MAX;
214  for (auto& solver : solvers_) solver->canceled_ = false;
215 
216  // run solvers_
217  {
218  BLOCKPROFILER("solve mt 2");
219  parallel_executor_->run();
220  }
221 
222  size_t best_index = 0;
223  best_fitness_ = DBL_MAX;
224 
225  // if exact primary goal matches have been found ...
226  for (size_t i = 0; i < thread_count_; ++i) {
227  if (solver_success_[i]) {
228  double fitness;
229  if (solvers_[0]->problem_.secondary_goals.empty()) {
230  // ... and if no secondary goals have been specified,
231  // select the best result_ according to primary goals
232  fitness = solver_fitness_[i];
233  } else {
234  // ... and if secondary goals have been specified,
235  // select the result_ that best satisfies primary and secondary goals
236  fitness = solver_fitness_[i] +
237  solvers_[0]->computeSecondaryFitnessAllVariables(
238  solver_solutions_[i]);
239  }
240  if (fitness < best_fitness_) {
241  best_fitness_ = fitness;
242  best_index = i;
243  }
244  }
245  }
246 
247  // if no exact primary goal matches have been found,
248  // select best primary goal approximation
249  if (best_fitness_ == DBL_MAX) {
250  for (size_t i = 0; i < thread_count_; ++i) {
251  if (solver_fitness_[i] < best_fitness_) {
252  best_fitness_ = solver_fitness_[i];
253  best_index = i;
254  }
255  }
256  }
257 
258  if (enable_counter_) {
259  LOG("iterations", iteration_count_);
260  }
261 
262  result_ = solver_solutions_[best_index];
263  success_ = solver_success_[best_index];
264  }
265 
266  double getSolutionFitness() const { return best_fitness_; }
267 
268  bool getSuccess() const { return success_; }
269 
270  const std::vector<double>& getSolution() const { return result_; }
271 };
