204 // starts profiler and periodically writes results to log
205 struct Profiler {
206  std::thread thread;
207  volatile int exit_flag;
208  Profiler() {
209  pthread_attr_t attr;
210  pthread_getattr_np(pthread_self(), &attr);
211  void* stack_addr;
212  size_t stack_size;
213  pthread_attr_getstack(&attr, &stack_addr, &stack_size);
214  profiler_info.stack_begin = stack_addr;
215  profiler_info.stack_end = (char*)stack_addr + stack_size;
216  const size_t maxbin = 1000;
217  static std::mutex mutex;
218  static std::unordered_map<const char*, size_t> samples;
219  exit_flag = 0;
220  std::thread t([this]() {
221  auto* profiler_bins = getProfilerBuffer();
222  while (true) {
223  for (int iter = 0; iter < 100; iter++) {
224  for (int iter = 0; iter < 100; iter++) {
225  int i = rand() % maxbin;
226  const char* p = profiler_bins[i].name;
227  if (p) samples[p]++;
228  }
229  if (exit_flag) break;
230  std::this_thread::sleep_for(
231  std::chrono::duration<size_t, std::micro>(rand() % 1000));
232  }
233  {
234  double thistime = rclcpp::Clock().now().seconds();
235  static double lasttime = 0.0;
236  if (thistime < lasttime + 1) continue;
237  lasttime = thistime;
238  static std::vector<std::pair<const char*, size_t>> data;
239  data.clear();
240  for (auto& p : samples) data.push_back(p);
241  std::sort(data.begin(), data.end(),
242  [](const std::pair<const char*, size_t>& a,
243  const std::pair<const char*, size_t>& b) {
244  return a.second > b.second;
245  });
246  LOG("");
247  LOG("profiler");
248  for (auto& d : data) {
249  double v = d.second * 100.0 / data[0].second;
250  char s[32];
251  sprintf(s, "%6.2f%%", v);
252  LOG("p", s, d.first);
253  }
254  LOG("");
255  }
256  if (exit_flag) break;
257  }
258  });
259  std::swap(thread, t);
260  }
261  ~Profiler() {
262  exit_flag = true;
263  thread.join();
264  }
265  static void start() { static Profiler profiler; }
266 };
