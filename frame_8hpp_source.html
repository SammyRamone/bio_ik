106 __attribute__((always_inline)) inline void quat_mul_vec(
107  const tf2::Quaternion& q, const tf2::Vector3& v, tf2::Vector3& r) {
108  double v_x = v.x();
109  double v_y = v.y();
110  double v_z = v.z();
111 
112  // if(__builtin_expect(v_x == 0 && v_y == 0 && v_z == 0, 0)) { r =

116  double q_x = q.x();
117  double q_y = q.y();
118  double q_z = q.z();

122  (q_x == 0 && q_y == 0 && q_z == 0 && q_w == 1)) {
123  r = v;
124  return;
125  }
126  // if((v_x + v_y + v_z == 0 && v_x == 0 && v_y == 0) || (q_x + q_y + q_z == 0
127  // && q_x == 0 && q_y == 0 && q_w == 1)) { r = v; return; } if(q_x == 0 && q_y
128  // == 0 && q_z == 0 && q_w == 1) { r = v; return; }

130  double t_x = q_y * v_z - q_z * v_y;
131  double t_y = q_z * v_x - q_x * v_z;
132  double t_z = q_x * v_y - q_y * v_x;

134  double r_x = q_w * t_x + q_y * t_z - q_z * t_y;
135  double r_y = q_w * t_y + q_z * t_x - q_x * t_z;
136  double r_z = q_w * t_z + q_x * t_y - q_y * t_x;

151 __attribute__((always_inline)) inline void quat_mul_quat(
152  const tf2::Quaternion& p, const tf2::Quaternion& q, tf2::Quaternion& r) {
153  double p_x = p.x();
154  double p_y = p.y();
155  double p_z = p.z();
156  double p_w = p.w();
157 
158  double q_x = q.x();
159  double q_y = q.y();
160  double q_z = q.z();
161  double q_w = q.w();
162 
163  double r_x = (p_w * q_x + p_x * q_w) + (p_y * q_z - p_z * q_y);
164  double r_y = (p_w * q_y - p_x * q_z) + (p_y * q_w + p_z * q_x);
165  double r_z = (p_w * q_z + p_x * q_y) - (p_y * q_x - p_z * q_w);
166  double r_w = (p_w * q_w - p_x * q_x) - (p_y * q_y + p_z * q_z);
167 
168  r.setX(r_x);
169  r.setY(r_y);
170  r.setZ(r_z);
171  r.setW(r_w);
172 }
