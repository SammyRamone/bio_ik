70  inline void getJointFrame(const moveit::core::JointModel* joint_model,
71  const double* variables, Frame& frame) {
72  auto joint_type = joint_model->getType();
73  if (joint_type == moveit::core::JointModel::FIXED) {
74  frame = Frame::identity();
75  return;
76  }
77  size_t joint_index = joint_model->getJointIndex();
78  switch (joint_type) {
79  case moveit::core::JointModel::REVOLUTE: {
80  auto axis = joint_axis_list_[joint_index];
81 
82  auto v = variables[joint_model->getFirstVariableIndex()];
83 
84  /*v *= 1.0 / (2 * M_PI);
85  v -= std::floor(v);
86  v *= (2 * M_PI);*/
87 
88  auto half_angle = v * 0.5;
89 
90  // TODO: optimize sin cos
91 
92  auto fcos = cos(half_angle);
93  auto fsin = sin(half_angle);
94 
95  // auto fsin = half_angle;
96  // auto fcos = 1.0 - half_angle * half_angle * 0.5;
97 
98  frame = Frame(Vector3(0.0, 0.0, 0.0),
99  Quaternion(axis.x() * fsin, axis.y() * fsin,
100  axis.z() * fsin, fcos));
101 
102  break;
103  }
104  case moveit::core::JointModel::PRISMATIC: {
105  auto axis = joint_axis_list_[joint_index];
106  auto v = variables[joint_model->getFirstVariableIndex()];
107  frame = Frame(axis * v, Quaternion(0.0, 0.0, 0.0, 1.0));
108  break;
109  }
110  case moveit::core::JointModel::FLOATING: {
111  auto* vv = variables + joint_model->getFirstVariableIndex();
112  frame.pos = Vector3(vv[0], vv[1], vv[2]);
113  frame.rot = Quaternion(vv[3], vv[4], vv[5], vv[6]).normalized();

115  // vv[2], vv[3], vv[4], vv[5], vv[6]);
116  break;
117  }
118  default: {
119  auto* joint_variables =
120  variables + joint_model->getFirstVariableIndex();
121  Eigen::Isometry3d joint_transform;
122  joint_model->computeTransform(joint_variables, joint_transform);
123  frame = Frame(joint_transform);
124  break;
125  }

561  void computeJacobian(const std::vector<size_t>& variable_indices,
562  Eigen::MatrixXd& jacobian) {
563  double step_size = 0.00001;
564  // double half_step_size = step_size * 0.5;
565  double inv_step_size = 1.0 / step_size;
566  auto tip_count = tip_frames.size();
567  jacobian.resize(static_cast<Eigen::Index>(tip_count * 6),
568  static_cast<Eigen::Index>(variable_indices.size()));
569  for (Eigen::Index icol = 0;
570  icol < static_cast<Eigen::Index>(variable_indices.size()); ++icol) {
571  for (Eigen::Index itip = 0; itip < static_cast<Eigen::Index>(tip_count);
572  ++itip) {
573  jacobian(itip * 6 + 0, icol) = 0;
574  jacobian(itip * 6 + 1, icol) = 0;
575  jacobian(itip * 6 + 2, icol) = 0;
576  jacobian(itip * 6 + 3, icol) = 0;
577  jacobian(itip * 6 + 4, icol) = 0;
578  jacobian(itip * 6 + 5, icol) = 0;
579  }
580  }
581  for (size_t icol = 0; icol < variable_indices.size(); icol++) {
582  auto ivar = variable_indices[icol];
583  auto* var_joint_model =
584  robot_model_->getJointOfVariable(static_cast<int>(ivar));
585  if (var_joint_model->getMimic()) continue;
586  for (auto* joint_model :
587  joint_dependencies[var_joint_model->getJointIndex()]) {
588  double scale = 1;
