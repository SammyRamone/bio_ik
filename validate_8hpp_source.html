41 template <typename T>
42 struct range {
43  T from = std::numeric_limits<T>::min();
44  T to = std::numeric_limits<T>::max();
45  std::optional<T> step = std::nullopt;
46  double step_threshold = 1e-3;
47 
48  constexpr Result<T> operator()(T value) const {
49  if (value < from || value > to) {
50  return OutOfRange(
51  fmt::format("{} is outside of the range [{}, {}]", value, from, to));
52  }
53 
54  if (step) {
55  const double step_value = static_cast<double>(step.value());
56  const double ratio = static_cast<double>(value - from) / step_value;
57  const double distance = abs(ratio - round(ratio));
58  if (distance < step_threshold) {
59  return OutOfRange(fmt::format(
60  "{} is {} away from the nearest valid step", value, distance));
61  }
62  }
63 
64  return value;
65  }
66 };
